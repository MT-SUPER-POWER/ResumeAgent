# V0.3 Concurrency Pipeline Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite pipeline as a 3-stage Producer-Consumer with Work-stealing, flume unbounded MPMC channel, and DB-backed result collection with failure-aware summary.

**Architecture:** Stage1 (M workers, Work-stealing) parses files and pushes to flume unbounded channel. Stage2 (N workers, Semaphore-controlled) pulls from channel, calls LLM, persists to DB. Stage3 queries DB via JOIN to collect per-run results and generate reports with failure lists.

**Tech Stack:** Rust, flume (MPMC), tokio::sync::Semaphore, tokio::spawn, SeaORM

**Reference:** `docs/architecture/pipeline-concurrency.md`

- [ ] **Step 2: Write new pipeline.rs**

```rust
use crate::errors::AppError;
use crate::llm::client::LlmClient;
use crate::llm::schemas::PipelineResult;
use crate::services::{evaluator, parser, reporter, summary};
use sha2::{Digest, Sha256};
use std::path::{Path, PathBuf};
use std::sync::Arc;
use std::sync::atomic::{AtomicUsize, Ordering};
use uuid::Uuid;

struct ParseTask {
    file_name: String,
    resume_id: i32,
    raw_text: String,
}

pub async fn run(files: &[PathBuf], cli_concurrency: Option<u32>) -> Result<(), AppError> {
    let cfg = crate::config::get();
    let m = cfg.pipeline.parse_workers;
    let n = cli_concurrency.unwrap_or_else(|| cfg.llm.concurrency.value());

    let timestamp = crate::repository::local_now()
        .format("%Y-%m-%dT%H-%M-%S").to_string();
    let run_dir = Path::new(&cfg.output.base_dir).join(&timestamp);
    tokio::fs::create_dir_all(&run_dir)
        .await
        .map_err(|e| AppError::Internal(format!("Cannot create output dir: {e}")))?;

    // DB pool check
    let needed = (m + n) * 2 + 2;
    if needed > cfg.db.max_connections {
        eprintln!("⚠ parse_workers={m}, concurrency={n}, (M+N)×2+2={needed} > max_connections={cur}, 建议调大 db.max_connections", cur = cfg.db.max_connections);
    }

    // Create job_run
    let run_id = Uuid::new_v4().to_string();
    let files_str: Vec<String> = files.iter().map(|p| p.to_string_lossy().to_string()).collect();
    let input_value = serde_json::to_string(&files_str).unwrap_or_default();
    crate::repository::job_run::create(&run_id, "files", &input_value, files.len() as i32).await?;
    println!("Job Run ID: {run_id}");
    println!("parse_workers={m}, concurrency={n}");

    // ===== flume channel (MPMC, unbounded) =====
    let (tx, rx) = flume::unbounded::<ParseTask>();

    // ===== Stage 1: Parse (M workers, Work-stealing) =====
    let idx = Arc::new(AtomicUsize::new(0));
    let parse_sem = Arc::new(tokio::sync::Semaphore::new(m as usize));
    let s1_run_id = run_id.clone();
    let mut s1_handles = vec![];
    let mut parsed_count: i32 = 0;
    let mut failed_count: i32 = 0;
    let mut errors_list: Vec<serde_json::Value> = Vec::new();

    loop {
        let permit = parse_sem.clone().acquire_owned().await;
        let v = idx.fetch_add(1, Ordering::Relaxed);
        if v >= files.len() { break; }

        let tx = tx.clone();
        let fp = files[v].clone();
        let rid = s1_run_id.clone();
        s1_handles.push(tokio::spawn(async move {
            let _permit = permit;
            let file_name = fp.file_name().unwrap().to_string_lossy();

            let content = match tokio::fs::read(&fp).await {
                Ok(c) => c,
                Err(e) => {
                    let _ = tx.send(ParseTask { file_name, resume_id: 0, raw_text: String::new() });
                    // failure tracked via pipeline_states, handled after parse
                    return (false, file_name, Some(serde_json::json!({
                        "file": file_name, "phase": "parse",
                        "reason": format!("Cannot read file: {e}"),
                    })));
                }
            };

            let mut hasher = Sha256::new();
            hasher.update(&content);
            let file_hash = format!("{:x}", hasher.finalize());
            let ext = fp.extension().and_then(|e| e.to_str()).unwrap_or("").to_lowercase();
            let file_type = if ext == "pdf" { "pdf" } else { "docx" };

            let existing = match crate::repository::resume::find_by_hash(&file_hash).await {
                Ok(r) => r,
                Err(e) => {
                    return (false, file_name.clone(), Some(serde_json::json!({
                        "file": file_name, "phase": "parse", "reason": format!("{e}"),
                    })));
                }
            };

            let (resume_id, raw_text) = if let Some(ref r) = existing {
                if r.parse_status == "success" && r.raw_text.is_some() {
                    (r.id, r.raw_text.clone())
                } else { (r.id, None) }
            } else {
                match crate::repository::resume::upsert(
                    &fp.to_string_lossy(), &file_hash, &file_name, file_type).await
                {
                    Ok(r) => (r.id, None),
                    Err(e) => {
                        return (false, file_name.clone(), Some(serde_json::json!({
                            "file": file_name, "phase": "parse", "reason": format!("{e}"),
                        })));
                    }
                }
            };

            let raw_text = if let Some(t) = raw_text { t } else {
                let parse_result = parser::parse_resume(&fp.to_string_lossy()).await;
                let _ = crate::repository::resume::update_parse(
                    resume_id, parse_result.raw_text.as_deref(),
                    &parse_result.status, parse_result.error.as_deref(),
                ).await;

                if parse_result.status != "success" {
                    let reason = parse_result.error.unwrap_or_else(|| "parse failed".into());
                    let _ = crate::repository::pipeline_state::upsert(
                        &rid, resume_id, "parse", "failed", Some(&reason),
                    ).await;
                    return (false, file_name.clone(), Some(serde_json::json!({
                        "file": file_name, "phase": "parse", "reason": reason,
                    })));
                }
                let _ = crate::repository::pipeline_state::upsert(
                    &rid, resume_id, "parse", "success", None,
                ).await;
                parse_result.raw_text.unwrap()
            };

            let _ = crate::repository::pipeline_state::upsert(
                &rid, resume_id, "parse", "success", None,
            ).await;

            let _ = tx.send(ParseTask { file_name: file_name.clone(), resume_id, raw_text });
            (true, file_name, None)
        }));
    }

    // Drop the original tx so Stage 2 can detect EOF when all Stage 1 tx clones are gone
    drop(tx);

    // ===== Stage 2: Evaluate (N workers, Semaphore-controlled, MPMC) =====
    let eval_sem = Arc::new(tokio::sync::Semaphore::new(n as usize));
    let client = Arc::new(LlmClient::new());
    let s2_run_id = run_id.clone();
    let mut s2_handles = vec![];

    for _ in 0..n {
        let rx = rx.clone();
        let sem = eval_sem.clone();
        let c = client.clone();
        let rid = s2_run_id.clone();
        s2_handles.push(tokio::spawn(async move {
            let mut success = 0i32;
            let mut failed = 0i32;
            loop {
                let task = match rx.recv_async().await {
                    Ok(t) => t,
                    Err(flume::RecvError::Disconnected) => break,
                };
                let permit = sem.clone().acquire_owned().await.expect("sem closed");
                let _permit = permit;
                match process_one_evaluate(task, &c, &rid).await {
                    (true, _, _) => success += 1,
                    (false, _, _) => failed += 1,
                }
            }
            (success, failed)
        }));
    }

    // ===== Await Stage 1 =====
    for h in s1_handles {
        match h.await {
            Ok((true, _, _)) => parsed_count += 1,
            Ok((false, _, Some(err))) => {
                failed_count += 1;
                errors_list.push(err);
            }
            _ => { failed_count += 1; }
        }
    }

    // ===== Await Stage 2 =====
    let mut evaluated_count: i32 = 0;
    for h in s2_handles {
        match h.await {
            Ok((s, f)) => { evaluated_count += s; failed_count += f; }
            Err(_) => { failed_count += 1; }
        }
    }

    // ===== Stage 3: Summary from DB =====
    let evals = crate::repository::evaluation::list_success_by_job_run(&run_id).await?;
    let mut rankings: Vec<summary::RankingEntry> = Vec::new();

    for e in &evals {
        let result: PipelineResult = e.result_json.as_ref()
            .and_then(|v| serde_json::from_value(v.clone()).ok())
            .unwrap_or_else(|| PipelineResult {
                candidate: crate::llm::schemas::CandidateInfo {
                    name: Some("未知".into()), email: None, phone: None, city: None,
                    years_of_experience: None, current_title: None,
                    education: vec![], skills: vec![], work_experience: vec![],
                    professional_titles: vec![],
                },
                target_role: None, best_match_role: None,
                evidence: Default::default(),
                talent_rating: crate::llm::schemas::TalentRating {
                    total_score: e.talent_score.unwrap_or(0.0), dimensions: vec![],
                },
                job_matching: crate::llm::schemas::JobMatching {
                    total_score: e.match_score.unwrap_or(0.0), dimensions: vec![],
                },
                overall_assessment: String::new(),
            });

        let name = result.candidate.name.clone().unwrap_or_else(|| "未知".into());
        let safe_name = name.replace(['<', '>', ':', '"', '/', '\\', '|', '?', '*'], "_");
        let report_md = reporter::generate_personal_report(&result);
        let report_path = reporter::save_personal_report(
            &run_dir.join("reports"), &name, &report_md,
        ).await.map_err(|e| AppError::Internal(format!("Save report: {e}")))?;

        let top_edu = result.candidate.education.first()
            .map(|ed| format!("{} {}", ed.school.as_deref().unwrap_or("-"), ed.degree.as_deref().unwrap_or("-")))
            .unwrap_or_else(|| "-".into());

        rankings.push(summary::RankingEntry {
            rank: 0, resume_id: e.resume_id, candidate_name: name,
            target_role: result.target_role, best_match_role: result.best_match_role,
            professional_titles: result.candidate.professional_titles,
            years_of_experience: result.candidate.years_of_experience,
            city: result.candidate.city, top_education: top_edu,
            talent_score: e.talent_score, match_score: e.match_score,
            skills: result.candidate.skills, overall_assessment: result.overall_assessment,
            report_path, report_link: format!("../reports/{safe_name}.md"),
        });
    }

    let failures: Vec<summary::FailureEntry> = crate::repository::pipeline_state::find_by_job_run(&run_id)
        .await.unwrap_or_default()
        .into_iter()
        .filter(|ps| ps.status == "failed")
        .map(|ps| summary::FailureEntry {
            file_name: format!("resume_id={}", ps.resume_id),
            phase: ps.phase,
            reason: ps.error_msg.unwrap_or_else(|| "unknown".into()),
        })
        .collect();

    if !rankings.is_empty() {
        rankings.sort_by(|a, b| b.talent_score.partial_cmp(&a.talent_score).unwrap());
        for (idx, e) in rankings.iter_mut().enumerate() { e.rank = idx + 1; }

        let title = format!("简历评估排名 — {}", timestamp);
        let ranking_md = summary::generate_ranking_md(&rankings, &failures, &title);
        let summary_json = summary::generate_summary_json(
            &run_id, &rankings, &failures,
            files.len() as i32, parsed_count, evaluated_count, failed_count, &title,
        );
        let (md_path, json_path) = summary::save_summary(
            &run_dir.join("summary"), &ranking_md, &summary_json,
        ).await.map_err(|e| AppError::Internal(format!("Save summary: {e}")))?;
        println!("Summary saved:\n  {md_path}\n  {json_path}");
    }

    if !errors_list.is_empty() {
        println!("\n===== FAILURES =====");
        for err in &errors_list {
            println!("  [{}] {} — {}",
                err["phase"].as_str().unwrap_or("-"),
                err["file"].as_str().unwrap_or("-"),
                err["reason"].as_str().unwrap_or("-"));
        }
        println!("====================");
    }

    let final_status = if failed_count > 0 && evaluated_count == 0 { "failed" } else { "completed" };
    crate::repository::job_run::update_progress(&run_id, parsed_count, evaluated_count, failed_count,
        if errors_list.is_empty() { None } else { Some(serde_json::to_value(&errors_list).unwrap()) },
    ).await?;
    crate::repository::job_run::mark_done(&run_id, final_status).await?;

    println!("\nDone! {} parsed, {} evaluated, {} failed (total: {})",
        parsed_count, evaluated_count, failed_count, files.len());
    Ok(())
}

async fn process_one_evaluate(
    task: ParseTask, client: &LlmClient, run_id: &str,
) -> (bool, String, Option<serde_json::Value>) {
    let existing_eval = match crate::repository::evaluation::find_by_resume(task.resume_id).await {
        Ok(Some(e)) if e.status == "success" => {
            let _ = crate::repository::pipeline_state::upsert(
                run_id, task.resume_id, "evaluate", "success", None).await;
            return (true, task.file_name, None);
        }
        Ok(_) => None,
        Err(e) => {
            let msg = format!("{e}");
            let _ = crate::repository::pipeline_state::upsert(
                run_id, task.resume_id, "evaluate", "failed", Some(&msg)).await;
            return (false, task.file_name.clone(), Some(serde_json::json!({
                "file": task.file_name, "phase": "evaluate", "reason": msg,
            })));
        }
    };

    match evaluator::evaluate(client, task.resume_id, &task.raw_text).await {
        Ok(_) => {
            let _ = crate::repository::pipeline_state::upsert(
                run_id, task.resume_id, "evaluate", "success", None).await;
            (true, task.file_name, None)
        }
        Err(e) => {
            let msg = format!("{e}");
            let _ = crate::repository::pipeline_state::upsert(
                run_id, task.resume_id, "evaluate", "failed", Some(&msg)).await;
            (false, task.file_name.clone(), Some(serde_json::json!({
                "file": task.file_name, "phase": "evaluate", "reason": msg,
            })))
        }
    }
}
```

```rust
Command::Rerun { id, concurrency } => {
    let job_run = if let Some(ref rid) = id {
        crate::repository::job_run::find_by_id(rid)
            .await?
            .ok_or_else(|| anyhow::anyhow!("Job run not found: {rid}"))?
    } else {
        crate::repository::job_run::find_latest()
            .await?
            .ok_or_else(|| anyhow::anyhow!(
                "No previous job run found. Run 'resume-agent run --dir <path>' first."
            ))?
    };

    println!("Rerunning job: {}", job_run.id);
    println!("  Input type: {}", job_run.input_type);
    println!("  Started at: {}", job_run.started_at);

    let files: Vec<PathBuf> = match job_run.input_type.as_str() {
        "files" => {
            let f: Vec<String> = serde_json::from_str(&job_run.input_value)
                .map_err(|e| anyhow::anyhow!("Failed to parse input_value: {e}"))?;
            f.into_iter().map(PathBuf::from).collect()
        }
        other => anyhow::bail!("Unknown input_type: {other}"),
    };

    let missing: Vec<String> = files.iter()
        .filter(|f| !f.exists())
        .map(|f| f.to_string_lossy().to_string())
        .collect();
    if !missing.is_empty() {
        anyhow::bail!("Some files no longer exist:\n  {}", missing.join("\n  "));
    }

    println!("  Files: {}", files.len());
    pipeline::run(&files, concurrency).await?;
```
