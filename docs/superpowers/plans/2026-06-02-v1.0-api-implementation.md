**Goal:** 将现有 CLI 所有命令抽象为 `/api/v1/*` REST API，加入 users 表 + JWT 认证 + SSE 实时推送，Actix-web 嵌入现有 Rust 项目。

**Architecture:** Actix-web HTTP server 作为新增入口（与 CLI 平行），JWT middleware 做认证 + 角色注入，handler 层调用现有 Service/Repository/Pipeline。SSE 通过 `tokio::sync::broadcast` 广播 Pipeline 状态快照。所有端点统一 `ApiResponse<T>` 信封。

**Tech Stack:** Rust, Actix-web 4, SeaORM, jsonwebtoken, bcrypt, tokio::sync::broadcast

        .await
        .map_err(AppError::DbError)

) -> Result<models::user::Model, AppError> {
    let now = super::local_now();
    let active = models::user::ActiveModel {
        id: Set(Uuid::new_v4()),
        username: Set(username.to_owned()),
        password_hash: Set(password_hash.to_owned()),
        role: Set("user".to_owned()),
        is_active: Set(true),
        created_at: Set(now),
        updated_at: Set(now),
    };
    active
        .insert(database::get())
        .await
        .map_err(AppError::DbError)
}

pub async fn seed_admin() -> Result<(), AppError> {
    // 只在表为空时插入 admin
    let count = models::user::Entity::find()
        .count(database::get())
        .await
        .map_err(AppError::DbError)?;
    if count > 0 {
        return Ok(());
    }

    let hash = bcrypt::hash("admin123", bcrypt::DEFAULT_COST)
        .map_err(|e| AppError::Internal(format!("bcrypt: {e}")))?;
    let now = super::local_now();
    let active = models::user::ActiveModel {
        id: Set(Uuid::new_v4()),
        username: Set("admin".to_owned()),
        password_hash: Set(hash),
        role: Set("admin".to_owned()),
        is_active: Set(true),
        created_at: Set(now),
        updated_at: Set(now),
    };
    active
        .insert(database::get())
        .await
        .map_err(AppError::DbError)?;
    Ok(())
}

pub async fn list(page: u64, page_size: u64) -> Result<(Vec<models::user::Model>, u64), AppError> {
    let paginator = models::user::Entity::find()
        .order_by_desc(models::user::Column::CreatedAt);
    let total = paginator
        .clone()
        .count(database::get())
        .await
        .map_err(AppError::DbError)?;
    let items = paginator
        .offset((page - 1) * page_size)
        .limit(page_size)
        .all(database::get())
        .await
        .map_err(AppError::DbError)?;
    Ok((items, total))
}

// src/errors/response.rs
use actix_web::HttpResponse;
use serde::Serialize;
use serde_json::Value;

#[derive(Serialize)]
pub struct ApiResponse<T: Serialize> {
    pub code: u16,
    pub msg: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub data: Option<T>,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub detail: Option<Value>,
}

impl<T: Serialize> ApiResponse<T> {
    pub fn ok(data: T) -> HttpResponse {
        HttpResponse::Ok().json(Self {
            code: 0,
            msg: "ok".into(),
            data: Some(data),
            detail: None,
        })
    }

    pub fn ok_empty() -> HttpResponse {
        HttpResponse::Ok().json(ApiResponse::<()> {
            code: 0,
            msg: "ok".into(),
            data: None,
            detail: None,
        })
    }
}

#[derive(Serialize)]
pub struct PageData<T: Serialize> {
    pub list: Vec<T>,
    pub page: u64,
    pub page_size: u64,
    pub total: u64,
}

impl<T: Serialize> ApiResponse<PageData<T>> {
    pub fn page(list: Vec<T>, page: u64, page_size: u64, total: u64) -> HttpResponse {
        ApiResponse::ok(PageData {
            list,
            page,
            page_size,
            total,
        })
    }
}

// src/utils/jwt.rs
use jsonwebtoken::{decode, encode, DecodingKey, EncodingKey, Header, Validation};
use serde::{Deserialize, Serialize};
use uuid::Uuid;

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct Claims {
    pub sub: String,       // user UUID
    pub username: String,
    pub role: String,
    pub exp: usize,
}

pub fn sign(user_id: Uuid, username: &str, role: &str, secret: &str, ttl_hours: u32) -> Result<String, Box<dyn std::error::Error>> {
    let exp = chrono::Utc::now()
        .checked_add_signed(chrono::Duration::hours(ttl_hours as i64))
        .unwrap()
        .timestamp() as usize;

    let claims = Claims {
        sub: user_id.to_string(),
        username: username.to_owned(),
        role: role.to_owned(),
        exp,
    };

    encode(
        &Header::default(),
        &claims,
        &EncodingKey::from_secret(secret.as_bytes()),
    )
    .map_err(|e| e.into())
}

pub fn verify(token: &str, secret: &str) -> Result<Claims, Box<dyn std::error::Error>> {
    let data = decode::<Claims>(
        token,
        &DecodingKey::from_secret(secret.as_bytes()),
        &Validation::default(),
    )?;
    Ok(data.claims)
}
```

// src/api/middleware.rs
use crate::errors::AppError;
use crate::utils::jwt;
use actix_web::{dev, Error, HttpMessage};
use actix_web::dev::{ServiceRequest, ServiceResponse};
use actix_web::middleware::Next;
use actix_web::body::EitherBody;

#[derive(Debug, Clone)]
pub struct AuthUser {
    pub id: uuid::Uuid,
    pub username: String,
    pub role: String,
}

pub async fn jwt_validator(
    req: ServiceRequest,
    next: Next<impl actix_web::body::MessageBody>,
) -> Result<ServiceResponse<EitherBody<impl actix_web::body::MessageBody>>, Error> {
    // 公开端点白名单
    let path = req.path();
    let is_public = path == "/api/v1/auth/register"
        || path == "/api/v1/auth/login";

    if is_public {
        return next.call(req).await.map(|r| r.map_into_left_body());
    }

    let auth_header = req
        .headers()
        .get("Authorization")
        .and_then(|v| v.to_str().ok())
        .and_then(|v| v.strip_prefix("Bearer "));

    let token = match auth_header {
        Some(t) => t,
        None => {
            let err = AppError::Forbidden;
            return Ok(req.into_response(err.error_response()).map_into_right_body());
        }
    };

    let cfg = crate::config::get();
    let claims = match jwt::verify(token, &cfg.auth.jwt_secret) {
        Ok(c) => c,
        Err(_) => {
            let err = AppError::Forbidden;
            return Ok(req.into_response(err.error_response()).map_into_right_body());
        }
    };

    let user_id = uuid::Uuid::parse_str(&claims.sub).unwrap_or_default();
    req.extensions_mut().insert(AuthUser {
        id: user_id,
        username: claims.username,
        role: claims.role,
    });

    next.call(req).await.map(|r| r.map_into_left_body())
}
```

// src/api/auth.rs
use actix_web::{web, HttpResponse};
use serde::Deserialize;
use uuid::Uuid;

use crate::errors::{AppError, response::ApiResponse};
use crate::repository;
use crate::utils::jwt;

#[derive(Deserialize)]
pub struct AuthBody {
    pub username: String,
    pub password: String,
}

#[derive(Serialize)]
pub struct TokenResponse {
    pub token: String,
}

// POST /api/v1/auth/register
pub async fn register(body: web::Json<AuthBody>) -> Result<HttpResponse, AppError> {
    if body.username.trim().is_empty() {
        return Err(AppError::MissingField("username".into()));
    }
    if body.password.trim().is_empty() {
        return Err(AppError::MissingField("password".into()));
    }

    // 唯一性检查
    if repository::user::find_by_username(&body.username).await?.is_some() {
        return Err(AppError::UsernameTaken(body.username.clone()));
    }

    let hash = bcrypt::hash(&body.password, bcrypt::DEFAULT_COST)
        .map_err(|e| AppError::Internal(format!("bcrypt: {e}")))?;

    let user = repository::user::create(&body.username, &hash).await?;
    let cfg = crate::config::get();
    let token = jwt::sign(user.id, &user.username, &user.role, &cfg.auth.jwt_secret, cfg.auth.jwt_ttl_hours)
        .map_err(|e| AppError::Internal(format!("jwt: {e}")))?;

    Ok(ApiResponse::ok(TokenResponse { token }))
}

// POST /api/v1/auth/login
pub async fn login(body: web::Json<AuthBody>) -> Result<HttpResponse, AppError> {
    let user = repository::user::find_by_username(&body.username)
        .await?
        .ok_or(AppError::InvalidCredentials)?;

    if !user.is_active {
        return Err(AppError::AccountDisabled);
    }

    Ok(ApiResponse::ok(TokenResponse { token }))
}

use actix_multipart::Multipart;
use actix_web::web;
use futures_util::StreamExt;
use std::path::PathBuf;

pub struct UploadedFile {
    pub file_name: String,
    pub path: PathBuf,
}

/// 解析 multipart 上传，将所有文件写入临时目录，返回路径列表
pub async fn save_uploads(
    mut payload: Multipart,
    job_id: &str,
) -> Result<Vec<UploadedFile>, crate::errors::AppError> {
    let dir = std::path::Path::new("/tmp/resume-agent/uploads").join(job_id);
    tokio::fs::create_dir_all(&dir)
        .await
        .map_err(|e| crate::errors::AppError::IoError(format!("create upload dir: {e}")))?;

    let mut files = vec![];
    while let Some(Ok(mut field)) = payload.next().await {
        let cd = field.content_disposition().clone();
        let file_name = cd
            .get_filename()
            .unwrap_or("unknown")
            .to_owned();
        let path = dir.join(&file_name);
        let mut f = tokio::fs::File::create(&path)
            .await
            .map_err(|e| crate::errors::AppError::IoError(format!("create file: {e}")))?;

        while let Some(Ok(chunk)) = field.next().await {
            f.write_all(&chunk)
                .await
                .map_err(|e| crate::errors::AppError::IoError(format!("write chunk: {e}")))?;
        }

        let ext = std::path::Path::new(&file_name)
            .extension()
            .and_then(|e| e.to_str())
            .unwrap_or("")
            .to_lowercase();

        if ext != "pdf" && ext != "docx" {
            return Err(crate::errors::AppError::UnsupportedFileType(file_name));
        }

        files.push(UploadedFile {
            file_name,
            path,
        });
    }

    if files.is_empty() {
        return Err(crate::errors::AppError::EmptyFileList);
    }
    Ok(files)
}

use serde::Serialize;

#[derive(Debug, Clone, Serialize)]
pub struct SseEvent {
    pub job_id: String,
    pub timestamp: String,
    pub stage: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub status: Option<String>,
    #[serde(skip_serializing_if = "Option::is_none")]

    pub message: Option<String>,
    #[serde(skip_serializing_if = "Option::is_none")]

    #[serde(skip_serializing_if = "Option::is_none")]
    pub code: Option<String>,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub parsed: Option<i32>,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub evaluated: Option<i32>,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub failed: Option<i32>,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub summary: Option<serde_json::Value>,
}

impl SseEvent {
    pub fn event_type(&self) -> &str {
        match self.stage.as_str() {
            "started" => "job_started",
            "stage" => "stage",
            "parsing" | "evaluating" => "progress",
            "warning" => "warning",
            "error" => "error",
            "heartbeat" => "heartbeat",
            "complete" => "complete",
            _ => "progress",
        }
    }

    pub fn new(job_id: &str, stage: &str) -> Self {
        Self {
            job_id: job_id.to_owned(),
            timestamp: chrono::Utc::now().to_rfc3339(),
            stage: stage.to_owned(),
            status: None,
            done: None,
            total: None,
            file: None,
            resume_id: None,
            message: None,
            attempt: None,
            code: None,
            parsed: None,
            evaluated: None,
            failed: None,
            summary: None,
        }
    }

    pub fn progress(job_id: &str, stage: &str, file: &str, resume_id: i32, done: usize, total: usize, parsed: i32, evaluated: i32, failed: i32) -> Self {
        Self {
            job_id: job_id.to_owned(),
            timestamp: chrono::Utc::now().to_rfc3339(),
            stage: stage.to_owned(),
            status: None,
            done: Some(done),
            total: Some(total),
            file: Some(file.to_owned()),
            resume_id: Some(resume_id),
            message: None,
            attempt: None,
            code: None,
            parsed: Some(parsed),
            evaluated: Some(evaluated),
            failed: Some(failed),
            summary: None,
        }
    }

    pub fn complete(job_id: &str, status: &str, summary: serde_json::Value) -> Self {
        Self {
            job_id: job_id.to_owned(),
            timestamp: chrono::Utc::now().to_rfc3339(),
            stage: "complete".to_owned(),
            status: Some(status.to_owned()),
            summary: Some(summary),
            done: None,
            total: None,
            file: None,
            resume_id: None,
            message: None,
            attempt: None,
            code: None,
            parsed: None,
            evaluated: None,
            failed: None,
        }
    }
}

static CHANNELS: std::sync::LazyLock<Mutex<HashMap<String, broadcast::Sender<SseEvent>>>> =
    std::sync::LazyLock::new(|| Mutex::new(HashMap::new()));

pub async fn register_channel(job_id: String, tx: broadcast::Sender<SseEvent>) {
    CHANNELS.lock().await.insert(job_id, tx);
}

pub async fn subscribe(job_id: &str) -> broadcast::Receiver<SseEvent> {
    let guard = CHANNELS.lock().await;
    guard.get(job_id).map(|tx| tx.subscribe()).unwrap_or_else(|| {
        // 返回一个已关闭的 channel，让 SSE 立即结束
        let (_, rx) = broadcast::channel(1);
        rx
    })
}

pub async fn remove_channel(job_id: &str) {
    CHANNELS.lock().await.remove(job_id);
}

// POST /api/v1/models (admin)
pub async fn add(
    req: HttpRequest,
    body: web::Json<serde_json::Value>,
) -> Result<HttpResponse, AppError> {
    require_admin(&req)?;
    // 复用现有 model add 逻辑（参考 cli/model.rs）
    let name = body["name"].as_str().unwrap_or("").to_string();
    let protocol = body["protocol"].as_str().unwrap_or("").to_lowercase();
    let api_key = body["api_key"].as_str().unwrap_or("").to_string();
    let input_price = body["input_price"].as_f64().unwrap_or(0.0);
    let output_price = body["output_price"].as_f64().unwrap_or(0.0);
    let base_url = body["base_url"].as_str().map(|s| s.to_string());

    if protocol != "openai" && protocol != "claude" {
        return Err(AppError::UnsupportedProtocol(protocol));
    }

    let cfg = crate::config::get();
    let mk = crate::utils::crypto::parse_master_key(&cfg.crypto.master_key)
        .map_err(|e| AppError::CryptoError(format!("{e}")))?;
    let encrypted = crate::utils::crypto::encrypt(&api_key, &mk)
        .map_err(|e| AppError::CryptoError(format!("{e}")))?;

    let active = crate::models::model_info::ActiveModel {
        name: sea_orm::Set(name.clone()),
        protocol: sea_orm::Set(protocol),
        input_price: sea_orm::Set(input_price),
        output_price: sea_orm::Set(output_price),
        encrypted_api_key: sea_orm::Set(encrypted),
        base_url: sea_orm::Set(base_url),
        enabled: sea_orm::Set(true),
    };
    repository::model_info::insert(active).await?;
    Ok(ApiResponse::ok_empty())
}

// POST /api/v1/users
pub async fn create(
    req: HttpRequest,
    body: web::Json<CreateUserBody>,
) -> Result<HttpResponse, AppError> {
    require_admin(&req)?;
    if repository::user::find_by_username(&body.username).await?.is_some() {
        return Err(AppError::UsernameTaken(body.username.clone()));
    }
    let hash = bcrypt::hash(&body.password, bcrypt::DEFAULT_COST)
        .map_err(|e| AppError::Internal(format!("bcrypt: {e}")))?;
    let user = repository::user::create(&body.username, &hash).await?;
    Ok(ApiResponse::ok(serde_json::json!({
        "id": user.id.to_string(), "username": user.username, "role": user.role,
    })))
}
